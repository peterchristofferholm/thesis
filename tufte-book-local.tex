% FORMATTING TITLES
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\titleformat{\chapter}%
	[display]%
    {\relax\begin{fullwidth}}%
    {\Large\sffamily \@chapapp{} \thechapter}
    {0em}
    {\huge\rmfamily\itshape}
	[\end{fullwidth}]

\titleformat{\part}%
	[display]%
    {\relax\begin{fullwidth}\centering}%
    {\Large PART \thepart}
    {0em}
    {\Huge\rmfamily\upshape}
	[\end{fullwidth}]

\titleformat{\section}%
  [hang]%
  {\normalfont\Large\itshape}%
  {\hspace{-3em}\makebox[3em]{%
    \normalsize\color{gray}\raggedleft\thesection}%
  }%
  {0em}%
  {}%
  []%

\titleformat{\subsection}%
  [hang]%
  {\normalfont\large\itshape}%
  {\hspace{-3em}\makebox[3em]{%
    \normalsize\color{gray}\raggedleft\thesubsection}%
  }%
  {0em}%
  {}%
  []%

\titlespacing*{\chapter}{0pt}{50pt}{40pt}
\titlespacing*{\section}{0pt}{3.5ex plus 1ex minus .2ex}{2.3ex plus .2ex}
\titlespacing*{\subsection}{0pt}{2.5ex plus 1ex minus .2ex}{1.5ex plus.2ex}
