\titleformat{\chapter}%
	[display]%
    {\relax\begin{fullwidth}}%
    {\Large\sffamily \@chapapp{} \thechapter}
    {0em}
    {\huge\rmfamily\itshape}
	[\end{fullwidth}]

\titleformat{\part}%
	[display]%
    {\relax\begin{fullwidth}\centering}%
    {\Large PART \thepart}
    {0em}
    {\Huge\rmfamily\upshape}
	[\end{fullwidth}]

\titleformat{\section}%
  [hang]%
  {\normalfont\Large\itshape}%
  {\hspace{-3em}\makebox[3em]{%
    \normalsize\color{gray}\raggedleft\thesection}%
  }%
  {0em}%
  {}%
  []%

\titleformat{\subsection}%
  [hang]%
  {\normalfont\large\itshape}%
  {\hspace{-3em}\makebox[3em]{%
    \normalsize\color{gray}\raggedleft\thesubsection}%
  }%
  {0em}%
  {}%
  []%
